  const [showPreview, setShowPreview] = useState(false);

          <h2>Preview</h2>

    </>
  );
}

// Add a fixed delay so you can see the loading state
function delayForDemo(promise) {
  return new Promise(resolve => {
    setTimeout(resolve, 2000);
  }).then(() => promise);
}
